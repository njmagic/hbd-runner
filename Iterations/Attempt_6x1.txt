  enemyGfx.henchman.background(0, 0, 0, 0);
  enemyGfx.henchman.noStroke();

  // Trenchcoat body
  enemyGfx.henchman.fill(30, 30, 35); // Dark fabric
  enemyGfx.henchman.rect(12, 15, 16, 25, 2); // Torso
  
  // Broad shoulders
  enemyGfx.henchman.rect(8, 15, 24, 8, 2); // Shoulders
  
  // Head
  enemyGfx.henchman.fill(30, 30, 35);
  enemyGfx.henchman.ellipse(20, 10, 12, 12);
  
  // Face in shadow
  enemyGfx.henchman.fill(15, 15, 20);
  enemyGfx.henchman.ellipse(20, 10, 10, 8);
  
  // Red eyes
  enemyGfx.henchman.fill(255, 0, 0);
  enemyGfx.henchman.ellipse(17, 10, 2, 2);
  enemyGfx.henchman.ellipse(23, 10, 2, 2);
  
  // Arms
  enemyGfx.henchman.fill(20, 20, 25);
  // Left arm
  enemyGfx.henchman.rect(6, 18, 6, 15, 2);
  // Right arm
  enemyGfx.henchman.rect(28, 18, 6, 15, 2);
  
  // Legs
  enemyGfx.henchman.fill(25, 25, 30);
  // Left leg
  enemyGfx.henchman.rect(14, 35, 6, 5, 1);
  // Right leg
  enemyGfx.henchman.rect(20, 35, 6, 5, 1);
  
  // Noir hat
  enemyGfx.henchman.fill(15, 15, 20);
  enemyGfx.henchman.arc(20, 6, 16, 8, PI, TWO_PI);
  enemyGfx.henchman.rect(12, 6, 16, 2);
  
  // Weapon - club or bat
  enemyGfx.henchman.fill(50, 50, 55);
  enemyGfx.henchman.rect(30, 22, 8, 4, 1);

  groundObstacleGfx.noStroke(); // Remove outlines

  groundObstacleGfx.fill(50, 50, 55); // Dark metal
  groundObstacleGfx.ellipse(16, 16, 20, 14); // Base of trash can
  groundObstacleGfx.rect(6, 16, 20, 16); // Bottom half of can
  
  // Metal rim details
  groundObstacleGfx.fill(70, 70, 75);
  groundObstacleGfx.rect(6, 16, 20, 2);
  groundObstacleGfx.rect(6, 26, 20, 2);
  
  // Trash spilling out
  groundObstacleGfx.fill(40, 40, 45);
  groundObstacleGfx.beginShape();
  groundObstacleGfx.vertex(6, 20);

  groundObstacleGfx.fill(80, 80, 85, 120);
  groundObstacleGfx.ellipse(18, 19, 3, 2);
  groundObstacleGfx.ellipse(12, 23, 2, 2);

  // Create flying obstacle graphics - spinning noir blades/broken ceiling fan
  flyingObstacleGfx = createGraphics(32, 32);
  flyingObstacleGfx.background(0, 0, 0, 0); // Transparent background
  flyingObstacleGfx.noStroke(); // Remove outlines

  flyingObstacleGfx.fill(60, 60, 65);
  flyingObstacleGfx.ellipse(16, 16, 10, 10);

  flyingObstacleGfx.fill(40, 40, 45);

function draw() {
  if (gameState === "start") {
    // Noir-style start screen
    background(0); // Pure black background
    
    // Rain effect
    drawRain();
    
    // Title with dramatic styling
    textAlign(CENTER);
    textSize(70);
    
    // Shadow/glow effect
    fill(255, 0, 0, 120);
    text("HBD", width / 2 + 4, height / 2 - 60 + sin(frameCount * 0.05) * 2);
    
    // Main text
    fill(255);
    text("HBD", width / 2, height / 2 - 60);
    
    // Tagline
    textSize(24);
    fill(200);
    text("HEROES BE DAMNED", width / 2, height / 2 - 20);
    
    // Instructions
    textSize(18);
    fill(255);
    text("Press SPACE to jump, DOWN to slide, Z to attack", width / 2, height / 2 + 40);
    
    // Start prompt with flickering
    if (frameCount % 30 < 20) {
      fill(255);
    } else {
      fill(255, 0, 0);
    }
    textSize(24);
    text("Press SPACE to start", width / 2, height / 2 + 80);
    
    // Dramatic lighting flash effect
    if (frameCount % 120 < 3) {
      // Lightning flash
      push();
      noStroke();
      fill(255, 30 + random(0, 20));
      rect(0, 0, width, height);
      pop();
    }
    
    if (keyIsDown(32)) { // 32 is the keyCode for SPACE
      // Reset game state
      gameState = "playing";
      score = 0;
      enemiesKilled = 0;
      enemies = [];
      obstacles = [];
      projectiles = [];
      shootEffects = [];
      slowMotion = false;
      slowMotionTimer = 0;
      scrollSpeed = normalScrollSpeed;
      screenFlash = 0;
      
      // Reset clouds to new random positions
      clouds = [];
      for (let i = 0; i < 5; i++) {
        clouds.push({
          x: random(0, width),
          y: random(20, 150),
          speed: random(0.3, 1.2),
          scale: random(0.5, 1.5),
          alpha: random(150, 230)
        });
      }
      
      player.y = groundY;
      player.vy = 0;
      player.state = "running";
      groundOffset = 0;
      backgroundOffset = 0;
      midgroundOffset = 0;
      foregroundOffset = 0;
    }
  } else if (gameState === "playing") {
    // Handle slow motion effect - we're disabling this by just keeping scrollSpeed at normal
    scrollSpeed = normalScrollSpeed;
    
    // Update scrolling ground and parallax backgrounds
    groundOffset += scrollSpeed;
    if (groundOffset > groundTile.width) {
      groundOffset -= groundTile.width;
    }

    // Update parallax offsets at different speeds
    backgroundOffset += scrollSpeed * 0.1; // Slowest (distant background)
    midgroundOffset += scrollSpeed * 0.4;  // Medium speed (midground)
    foregroundOffset += scrollSpeed * 0.7; // Faster (foreground, but still slower than ground)
    
    // Wrap offsets when they exceed the layer width
    if (backgroundOffset >= backgroundGfx.width) backgroundOffset -= backgroundGfx.width;
    if (midgroundOffset >= midgroundGfx.width) midgroundOffset -= midgroundGfx.width;
    if (foregroundOffset >= foregroundGfx.width) foregroundOffset -= foregroundGfx.width;

    // Draw gradient sky background - pure black to very dark gray
    background(0);
    
    // Draw stars - white dots
    drawTwinklingStars();
    
    // Draw background city skyline
    image(backgroundGfx, -backgroundOffset, 0);
    if (backgroundOffset > backgroundGfx.width - width) {
      image(backgroundGfx, -backgroundOffset + backgroundGfx.width, 0);
    }
    
    // Draw stationary moon - draw it here to be fixed in position
    drawMoon(600, 80);
    
    // Add occasional lightning flash in background
    if (random() < 0.002) {
      push();
      noStroke();
      fill(255, 255, 255, 30);
      rect(0, 0, width, height);
      pop();
    }
    
    // Update and draw clouds
    for (let i = 0; i < clouds.length; i++) {
      let cloud = clouds[i];
      cloud.x -= cloud.speed;
      
      // Wrap clouds around when they move off-screen
      if (cloud.x < -100 * cloud.scale) {
        cloud.x = width + random(0, 100);
        cloud.y = random(20, 150);
      }
      
      // Draw cloud with varying opacity and scale
      push();
      tint(255, cloud.alpha);
      image(cloudGfx, cloud.x, cloud.y, 
            cloudGfx.width * cloud.scale, 
            cloudGfx.height * cloud.scale);
      noTint();
      pop();
    }
    
    // Light rain effect in the background
    drawRain();
    
    // Draw midground buildings (wrapping around when needed)
    image(midgroundGfx, -midgroundOffset, 0);
    if (midgroundOffset > midgroundGfx.width - width) {
      image(midgroundGfx, -midgroundOffset + midgroundGfx.width, 0);
    }
    
    // Draw foreground buildings (wrapping around when needed)
    image(foregroundGfx, -foregroundOffset, 0);
    if (foregroundOffset > foregroundGfx.width - width) {
      image(foregroundGfx, -foregroundOffset + foregroundGfx.width, 0);
    }
    
    // Draw street fog - grayscale only (reduced amount)
    drawStreetFog();
    
    // Draw a simple continuous floor at ground level - adjust position
    let xOffset = -(groundOffset % groundTile.width);
    for (let x = xOffset - 128; x < width + 128; x += groundTile.width) {
      // Position the floor so its top edge aligns with groundY
      image(groundTile, x, groundY);
    }
    
    // Add a solid floor line where player's feet rest
    push();
    stroke(60);
    strokeWeight(1);
    line(0, groundY, width, groundY);
    pop();
    
    // Draw street reflections - white only (reduced)
    drawStreetReflections();

    // Update player position (jumping)
    if (player.state === "jumping") {
      player.y += player.vy;
      player.vy += gravity;
      if (player.y >= groundY) {
        player.y = groundY;
        player.vy = 0;
        player.state = "running";
      }
    }

    // Update player animation
    if (player.state === "running") {
      player.animTimer++;
      if (player.animTimer >= 5) {
        player.animTimer = 0;
        player.animFrame = (player.animFrame + 1) % 4;
      }
    }

    // Draw player shadow
    drawPlayerShadow();

    // Draw player based on state
    if (player.state === "running") {
      image(playerRunningFrames[player.animFrame], player.x, player.y - playerRunningFrames[player.animFrame].height);
    } else if (player.state === "jumping") {
      image(playerJumpingFrame, player.x, player.y - playerJumpingFrame.height);
    } else if (player.state === "ducking") {
      image(playerDuckingFrame, player.x, player.y - playerDuckingFrame.height);
    }

    // Update and draw enemies with enhanced size and appeal
    for (let i = enemies.length - 1; i >= 0; i--) {
      let enemy = enemies[i];
      
      // Apply different movement patterns based on enemy type
      if (enemy.type === 'flying') {
        // Flying enemy moves in a wave pattern
        enemy.x -= scrollSpeed;
        enemy.y = enemy.baseY + sin(frameCount * 0.05 + enemy.offset) * 40; // Smoother, larger wave pattern
      } else if (enemy.type === 'henchman') {
        // Henchman enemy moves slower but steadily
        enemy.x -= scrollSpeed * 0.7;
      } else if (enemy.type === 'chase') {
        // Chasing enemy moves faster but stays on ground level
        enemy.x -= scrollSpeed * 1.2;
        enemy.y = groundY; // Always stay at ground level
      } else {
        // Regular enemy
        enemy.x -= scrollSpeed;
      }
      
      if (enemy.x < -40) {
        enemies.splice(i, 1);
      } else {
        // Draw shadows for ground-based enemies - larger and more dramatic
        if (enemy.type !== 'flying') {
          push();
          noStroke();
          fill(0, 0, 0, 40);
          if (enemy.type === 'henchman') {
            ellipse(enemy.x + 25, groundY + 2, 45, 6); // Larger shadow for henchman
          } else {
            ellipse(enemy.x + 16, groundY + 2, 30, 4); // Slightly larger shadow for regular
          }
          pop();
        }
        
        // Draw based on enemy type - position all at ground level with enhanced size
        if (enemy.type === 'flying') {
          // Draw flying enemy at 1.25x scale
          push();
          // Remove red glow effect
          
          // Draw actual enemy image larger
          imageMode(CENTER);
          image(enemyGfx.flying, enemy.x + 20, enemy.y + 20, enemyGfx.flying.width * 1.25, enemyGfx.flying.height * 1.25);
          imageMode(CORNER);
          pop();
          
          // Flying enemy shadow (fainter and stretched based on height)
          push();
          noStroke();
          let shadowSize = map(enemy.y, groundY - 150, groundY, 15, 30);
          let shadowAlpha = map(enemy.y, groundY - 150, groundY, 10, 40);
          fill(0, 0, 0, shadowAlpha);
          ellipse(enemy.x + 20, groundY + 2, shadowSize, 3);
          pop();
        } else if (enemy.type === 'henchman') {
          // Draw henchman enemy
          push();
          
          // Draw health indicators based on remaining health
          let henchmanGfx = createGraphics(enemyGfx.henchman.width, enemyGfx.henchman.height);
          henchmanGfx.image(enemyGfx.henchman, 0, 0);
          
          // Add health indicators
          henchmanGfx.noStroke();
          henchmanGfx.fill(255, 0, 0);
          
          // Different health indicators based on health
          if (enemy.health >= 3) {
            // Full health - three red dots
            henchmanGfx.ellipse(10, 15, 4, 4);
            henchmanGfx.ellipse(20, 15, 4, 4);
            henchmanGfx.ellipse(30, 15, 4, 4);
          } else if (enemy.health === 2) {
            // Two health - two red dots
            henchmanGfx.ellipse(15, 15, 4, 4);
            henchmanGfx.ellipse(25, 15, 4, 4);
          } else if (enemy.health === 1) {
            // One health - one red dot
            henchmanGfx.ellipse(20, 15, 4, 4);
          }
          
          // Draw henchman enemy 1.5x scale
          imageMode(CENTER);
          image(henchmanGfx, enemy.x + 20, groundY - enemyGfx.henchman.height/2, enemyGfx.henchman.width * 1.5, enemyGfx.henchman.height * 1.5);
          imageMode(CORNER);
          pop();
        } else if (enemy.type === 'chase') {
          // Remove threatening glow for chase enemy
          push();
          // Draw chase enemy at 1.25x scale
          imageMode(CENTER);
          image(enemyGfx.regular, enemy.x + 16, enemy.y - enemyGfx.regular.height/2, enemyGfx.regular.width * 1.25, enemyGfx.regular.height * 1.25);
          imageMode(CORNER);
          pop();
        } else {
          // Remove subtle glow for regular enemy
          push();
          // Draw regular enemy 1.2x scale
          imageMode(CENTER);
          image(enemyGfx.regular, enemy.x + 16, groundY - enemyGfx.regular.height/2, enemyGfx.regular.width * 1.2, enemyGfx.regular.height * 1.2);
          imageMode(CORNER);
          pop();
        }
      }
    }

    // Update and draw obstacles
    for (let i = obstacles.length - 1; i >= 0; i--) {
      let obstacle = obstacles[i];
      obstacle.x -= scrollSpeed;
      if (obstacle.x < -32) {
        obstacles.splice(i, 1);
      } else {
        // Draw shadows for obstacles
        if (obstacle.type === "ground") {
          fill(0, 0, 0, 70);
          ellipse(obstacle.x + 16, groundY + 2, 28, 4);
          image(groundObstacleGfx, obstacle.x, groundY - groundObstacleGfx.height);
        } else if (obstacle.type === "flying") {
          // Flying obstacle shadow
          let shadowSize = map(obstacle.y, groundY - 150, groundY, 10, 24);
          let shadowAlpha = map(obstacle.y, groundY - 150, groundY, 20, 60);
          fill(0, 0, 0, shadowAlpha);
          ellipse(obstacle.x + 16, groundY + 2, shadowSize, 3);
          
          image(flyingObstacleGfx, obstacle.x, obstacle.y);
        }
      }
    }

    // Update and draw projectiles
    for (let i = projectiles.length - 1; i >= 0; i--) {
      let projectile = projectiles[i];
      projectile.x += 10; // Projectile speed
      if (projectile.x > width) {
        projectiles.splice(i, 1);
      } else {
        image(projectileGfx, projectile.x, projectile.y - 4);
      }
    }

    // Update and draw shoot effects
    for (let i = shootEffects.length - 1; i >= 0; i--) {
      let effect = shootEffects[i];
      
      // If this is an enemy defeat particle
      if (effect.vx !== undefined && effect.vy !== undefined) {
        effect.x += effect.vx;
        effect.y += effect.vy;
      } else {
        effect.size += 2;
      }
      
      effect.alpha -= 25;
      effect.life--;
      
      if (effect.life <= 0) {
        shootEffects.splice(i, 1);
      } else {
        push();
        noStroke();
        
        // If it's an enemy defeat effect, use red for blood splatter
        if (effect.vx !== undefined) {
          if (effect.isBlood) {
            // Draw irregular blood splatter - red only
            fill(255, 0, 0, effect.alpha);
            
            // Irregular shape for blood splatter
            beginShape();
            let splatterPoints = 6;
            for (let p = 0; p < splatterPoints; p++) {
              let ang = p * TWO_PI / splatterPoints;
              let rad = effect.size * (0.5 + noise(effect.x + p, effect.y, frameCount * 0.05) * 0.8);
              vertex(effect.x + cos(ang) * rad, effect.y + sin(ang) * rad);
            }
            endShape(CLOSE);
          } else {
            // Normal explosion particle - white only
            fill(255, effect.alpha);
            ellipse(effect.x, effect.y, effect.size, effect.size);
          }
        } else {
          // White and red for projectile impact
          if (effect.size % 4 < 2) {
            fill(255, effect.alpha);
          } else {
            fill(255, 0, 0, effect.alpha);
          }
          ellipse(effect.x, effect.y, effect.size, effect.size);
        }
        
        pop();
      }
    }

    // Collision detection - complete rewrite for safer approach
    let playerHitbox = getPlayerHitbox();
    
    // Check collision with obstacles
    for (let i = obstacles.length - 1; i >= 0; i--) {
      let obstacle = obstacles[i];
      let obstacleHitbox = getObstacleHitbox(obstacle);
      if (checkCollision(playerHitbox, obstacleHitbox)) {
        gameState = "gameOver";
        break; // Exit loop if game over
      }
    }
    
    // Only check enemy collisions if game is still in playing state
    if (gameState === "playing") {
      // First check if player collides with any enemy
      for (let i = enemies.length - 1; i >= 0; i--) {
        let enemy = enemies[i];
      let enemyHitbox = getEnemyHitbox(enemy);
        
      if (checkCollision(playerHitbox, enemyHitbox)) {
        gameState = "gameOver";
          break; // Exit loop if game over
        }
      }
    }
    
    // Separate loop for projectile collisions to avoid modifying the array while iterating
    if (gameState === "playing") {
      // Collect indices to remove rather than splicing immediately
      let enemyIndices = [];
      let projectileIndices = [];
      
      // Check all projectile-enemy collisions
      for (let j = 0; j < projectiles.length; j++) {
        let projectile = projectiles[j];
        let projectileHitbox = { x: projectile.x, y: projectile.y - 4, width: 8, height: 8 };
        let hitEnemy = false;
        
        for (let i = 0; i < enemies.length; i++) {
          let enemy = enemies[i];
          let enemyHitbox = getEnemyHitbox(enemy);
          
          // If collision detected
        if (checkCollision(projectileHitbox, enemyHitbox)) {
            // Mark projectile for removal
            if (!projectileIndices.includes(j)) {
              projectileIndices.push(j);
            }
            
            // For henchman enemies, reduce health instead of immediately removing
            if (enemy.type === 'henchman') {
              enemy.health--;
              
              // Create hit effect
              createEnemyHitEffect(enemy);
              
              // Only mark henchman for removal if health is depleted
              if (enemy.health <= 0) {
                createEnemyDefeatEffects(enemy);
                if (!enemyIndices.includes(i)) {
                  enemyIndices.push(i);
                }
                enemiesKilled++;
                score += 300; // Bonus points for henchman
                
                // Add screen flash effect only, without slow motion
                screenFlash = 30;
              }
            } else {
              // Regular enemies die in one hit
              createEnemyDefeatEffects(enemy);
              if (!enemyIndices.includes(i)) {
                enemyIndices.push(i);
              }
              enemiesKilled++;
              
              // Award points based on enemy type
              if (enemy.type === 'flying') {
                score += 200;
              } else if (enemy.type === 'chase') {
                score += 150;
              } else {
                score += 100;
              }
            }
            
            hitEnemy = true;
            break; // A projectile can only hit one enemy
          }
        }
      }
      
      // Remove projectiles (in reverse order to maintain correct indices)
      projectileIndices.sort((a, b) => b - a);
      for (let i = 0; i < projectileIndices.length; i++) {
        projectiles.splice(projectileIndices[i], 1);
      }
      
      // Remove enemies (in reverse order to maintain correct indices)
      enemyIndices.sort((a, b) => b - a);
      for (let i = 0; i < enemyIndices.length; i++) {
        enemies.splice(enemyIndices[i], 1);
      }
    }

    // Spawn new enemies and obstacles
    // Calculate spawn probability based on score to increase difficulty
    let difficultyFactor = 1 + (score / 5000); // Gradually increase difficulty
    let groundObstacleChance = 0.005 * difficultyFactor;
    let flyingObstacleChance = 0.005 * difficultyFactor;
    let enemyChance = 0.003 * difficultyFactor;
    
    // Check if we can spawn an obstacle (not too close to obstacles)
    let minDistanceBetweenObstacles = 150;
    let canSpawnObstacle = true;
    
    for (let obstacle of obstacles) {
      if (obstacle.x > width - minDistanceBetweenObstacles) {
        canSpawnObstacle = false;
        break;
      }
    }
    
    if (canSpawnObstacle) {
      if (random() < groundObstacleChance) {
        obstacles.push({ x: width + random(0, 50), y: groundY, type: "ground" });
      }
      if (random() < flyingObstacleChance) {
        obstacles.push({ x: width + random(0, 50), y: groundY - 100 + random(-30, 30), type: "flying" });
      }
    }
    
    // Check if we can spawn an enemy (not too close to obstacles)
    let canSpawnEnemy = true;
    for (let obstacle of obstacles) {
      if (obstacle.x > width - minDistanceBetweenObstacles) {
        canSpawnEnemy = false;
        break;
      }
    }
    
    if (canSpawnEnemy && random() < enemyChance) {
      // Decide which type of enemy to spawn based on score and randomness
      let enemyRoll = random();
      
      if (score > 2000 && enemyRoll < 0.2) {
        // Henchman enemy - 20% chance after score 2000
        enemies.push({
          x: width + random(0, 50),
          y: groundY, // Ground level 
          type: 'henchman',
          health: 3 // Henchman takes 3 hits to defeat
        });
      } else if (score > 1000 && enemyRoll < 0.4) {
        // Flying enemy - 20% chance after score 1000
        enemies.push({
          x: width + random(0, 50),
          y: groundY - 100, // Flying height - higher up from ground
          baseY: groundY - 100, // Base height for wave pattern
          offset: random(0, TWO_PI),
          type: 'flying'
        });
      } else if (score > 500 && enemyRoll < 0.6) {
        // Chasing enemy - 20% chance after score 500
        enemies.push({
          x: width + random(0, 50),
          y: groundY, // Ground level
          type: 'chase'
        });
      } else {
        // Regular enemy - always available
        enemies.push({
          x: width + random(0, 50),
          y: groundY, // Ground level
          type: 'regular'
        });
      }
    }

    // Update and display score with noir styling
    score += 1;
    push();
    fill(255);
    textSize(18);
    textAlign(LEFT);
    text("SCORE: " + score, 20, 30);
    pop();
    
    // Render screen flash effect - keep RED
    if (screenFlash > 0) {
      push();
      noStroke();
      // Fade from red to transparent
      fill(255, 0, 0, map(screenFlash, 30, 0, 150, 0));
      rect(0, 0, width, height);
      screenFlash--;
      pop();
    }
  } else if (gameState === "gameOver") {
    // Noir-style game over screen - pure black
    background(0);
    
    // Slow rain effect
    drawSlowRain();
    
    // Dramatic lighting
    push();
    noStroke();
    fill(255, 15);
    // Spotlight effect
    let spotlightSize = 300 + sin(frameCount * 0.05) * 20;
    ellipse(width/2, height/2, spotlightSize, spotlightSize);
    pop();
    
    // Game over text
    push();
    textAlign(CENTER);
    
    // Shadow
    textSize(64);
    fill(255, 0, 0);
    text("GAME OVER", width / 2 + 4, height / 2 - 60);
    
    // Main text
    fill(255);
    text("GAME OVER", width / 2, height / 2 - 65);
    pop();
    
    // Score display
    textSize(32);
    fill(255);
    text("SCORE: " + score, width / 2, height / 2);
    
    // Display enemies killed count
    textSize(24);
    fill(220);
    text("ENEMIES DEFEATED: " + enemiesKilled, width / 2, height / 2 + 40);
    
    // Flavor text
    push();
    fill(180);
    textSize(18);
    text("THE CITY CLAIMS ANOTHER HERO...", width / 2, height / 2 + 80);
    pop();
    
    // Restart prompt with flickering effect
    if (frameCount % 30 < 20) {
      fill(255);
    } else {
      fill(255, 0, 0);
    }
    textSize(24);
    text("PRESS R TO FIGHT AGAIN", width / 2, height / 2 + 120);
    
    // Dramatic lighting flash
    if (frameCount % 180 < 3) {
      push();
      noStroke();
      fill(255, 30);
      rect(0, 0, width, height);
      pop();
    }
    
    if (keyIsDown(82)) { // 82 is the keyCode for 'R'
      gameState = "start";
    }
  }
}

// Helper function to draw player shadow - pure black

  push();
  noStroke();

  if (player.state === "jumping") {

    let shadowSize = map(player.y, groundY - 150, groundY, 15, 30);

    ellipse(player.x + 16, groundY + 2, 30, 4); // Slightly below ground line
  }
  
  pop();
}

// Helper functions for collision detection
function getPlayerHitbox() {
  let height = player.state === "ducking" ? playerDuckingFrame.height : playerJumpingFrame.height;
  return { x: player.x, y: player.y - height, width: 32, height: height };
}

function getObstacleHitbox(obstacle) {
  let height = obstacle.type === "ground" ? groundObstacleGfx.height : flyingObstacleGfx.height;
  let y = obstacle.type === "ground" ? obstacle.y - height : obstacle.y;
  return { x: obstacle.x, y: y, width: 32, height: height };
}

function getEnemyHitbox(enemy) {
  if (enemy.type === 'flying') {
    // Flying enemy hitbox
    return { x: enemy.x, y: enemy.y - 16, width: 32, height: 16 };
  } else if (enemy.type === 'henchman') {
    // Henchman enemy hitbox
    return { x: enemy.x, y: groundY - 40, width: 40, height: 40 };
  } else if (enemy.type === 'chase') {
    // Chase enemy hitbox
    return { x: enemy.x, y: enemy.y - 32, width: 32, height: 32 };
  } else {
    // Regular enemy hitbox
    return { x: enemy.x, y: groundY - 32, width: 32, height: 32 };
  }
}

function checkCollision(rect1, rect2) {
  return rect1.x < rect2.x + rect2.width &&
         rect1.x + rect1.width > rect2.x &&
         rect1.y < rect2.y + rect2.height &&
         rect1.y + rect1.height > rect2.y;
}

// Handle user input
function keyPressed() {
  if (gameState === "playing") {
    if (key === ' ' || keyCode === UP_ARROW) {
      if (player.state === "running") {
        player.vy = jumpStrength;
        player.state = "jumping";
      }
    } else if (keyCode === DOWN_ARROW) {
      if (player.state === "running") {
        player.state = "ducking";
      }
    } else if (key === 'z' || key === 'Z') {
      projectiles.push({ x: player.x + 32, y: player.y - 16 });
      
      // Add a brief visual feedback effect when shooting
      let shootEffect = {
        x: player.x + 32, 
        y: player.y - 16,
        size: 12,
        alpha: 255,
        life: 10
      };
      
      shootEffects.push(shootEffect);
    }
  }
}

function keyReleased() {
  if (keyCode === DOWN_ARROW) {
    if (player.state === "ducking") {
      player.state = "running";
    }
  }
}

// Visual effect for enemy defeat - enhance the death effects
function createEnemyDefeatEffects(enemy) {
  // More particles for enemy defeat
  for (let k = 0; k < 12; k++) {
    let angle = k * (PI/6);
    let speed = random(1.5, 3);
    let effect = {
      x: enemy.x + 16,
      y: enemy.y - 16,
      vx: cos(angle) * speed,
      vy: sin(angle) * speed,
      size: random(6, 10),
      alpha: 255,
      life: 25
    };
    shootEffects.push(effect);
  }
  
  // More dramatic blood splatter
  for (let k = 0; k < 6; k++) {
    let angle = random(0, TWO_PI);
    let distance = random(10, 30);
    let splatter = {
      x: enemy.x + 16 + cos(angle) * distance,
      y: enemy.y - 16 + sin(angle) * distance,
      vx: cos(angle) * 0.5,
      vy: sin(angle) * 0.5 + 0.2, // More downward drift
      size: random(5, 15),
      alpha: 255,
      life: 35,
      isBlood: true
    };
    shootEffects.push(splatter);
  }
  
  // Add a flash of light
  let flash = {
    x: enemy.x + 16,
    y: enemy.y - 16,
    size: 5,
    alpha: 200,
    life: 15
  };
  shootEffects.push(flash);
}
